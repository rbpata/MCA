#Code :-
#include<stdio.h>
int main(){

    int arr[20],no,i;

    printf("\nHow Many Elements :- ");
    scanf("%d",&no);
    for(i=0;i<no;i++){
        printf("\nEnter %d element :-",i+1);
        scanf("%d",&arr[i]);
    }
    printf("\nPrinting Array Element in Reverse Order...");
    for(i=no-1;i>=0;i--){
        printf("\n%d Element is :- %d",i+1,arr[i]);
    }

    return 0;
}

#Output :-

How Many Elements :- 5

Enter 1 element :-1

Enter 2 element :-2

Enter 3 element :-3

Enter 4 element :-4

Enter 5 element :-5

Printing Array Element in Reverse Order...
5 Element is :- 5
4 Element is :- 4
3 Element is :- 3
2 Element is :- 2
1 Element is :- 1
